
x = 7 = 0000 0111 In Bin

y = 4 = 0000 0100 In Bin


1 - z = x & y 

{
   z = 0000 0111 
     & 0000 0100 
   z = 0000 0100
   z = 4
}

2 - k = x | y

{
    k = 0000 0111  
      | 0000 0100  
    k = 0000 0111
    k = 7
}

3 - m = x ^ y

{
   m = 0000 0111 
     ^ 0000 0100
   m = 0000 0011
   m = 3
} 

4 - l = x >> 1

{
   l = 0000 0111 >> 1
   l = 0000 0011
   l = 3
}

5 - n = y << 2

{
   n = 0000 0100 << 2
   n = 0001 0000
   n = 16
}
